function pivot(list, left, right)

    pivotValue := list[pivotIndex]
    swap(list[pivotIndex], list[right])  // Move pivot to end

function insertionSort(list, left, right)
    for i from left + 1 to right
        key := list[i]
        j := i - 1
        while j >= left and list[j] > key
            list[j + 1] := list[j]
