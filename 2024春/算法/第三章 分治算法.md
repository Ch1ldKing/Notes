Algorithm FindMinMax(A)
    Input: An array A of n elements
    Output: The minimum and maximum of A

    if n mod 2 == 0 then
        if A[1] > A[2] then
            max = A[1]
            min = A[2]
        else
            max = A[2]
            min = A[1]
        end if
        i = 3  // Start from the third element
    else
        max = A[1]
        min = A[1]
        i = 2  // Start from the second element
    end if

    while i < n do
        if A[i] > A[i + 1] then
            if A[i] > max then
                max = A[i]
            end if
            if A[i + 1] < min then
                min = A[i + 1]
            end if
        else
            if A[i + 1] > max then
                max = A[i + 1]
            end if
            if A[i] < min then
                min = A[i]
            end if
        end if
        i = i + 2  // Move to the next pair
    end while

    return min, max
